      • + - HTTP

        • import http = require("http");

          + - createServer(requestListener?: ( request: IncomingMessage, response: ServerResponse ) => void ): Server;

          • creates and returns an instance of Server

          + - Server

          • + - listen ( port: number ): Server;
            • tells the serverinstance to listen on the given port

          + - IncomingMessage

          • created automatically by the server and sent as the first parameter to the requestListener
          • + - method: string
            • "GET", "POST", etc.
          • + - url: string
            • the path including the search-string the server was requested with

          + - ServerResponse

          • created automatically by the server and sent as the second parameter to the requestListener
          • + - setHeader ( name: string, value: string | string[ ] ): void;
            • writes a header to the response stream
            • e.g. setHeader("content-type", "text/html")
          • + - write ( text: string, callback?: Function ): boolean;
            • This sends a string to the response body
            • This method may be called multiple times to provide successive parts of the body.
            • returns true, when the string was successfully integrated
            • by default, the encoding is utf-8
          • + - end ( text: string, callback?: Function ): void;
            • same as write, but completes the response body
            • and signals the server to send the response to the client
            • Thus, this method must be called on each response!
            • no return value

  • + - MongoDB
    • shell commands
    • + - node classes and interfaces
      • as defined in module "mongodb"
      • + - MongoCallback
        • Many methods require a callback function as parameter
        • + - of type MongoCallback<T>
          • function (error: MongoError, result: T): void;
          • where T is the type of the actual data
      • + - MongoClient
        • Connect to MongoDB using a url of a database
        • connect (url: string, callback: MongoCallback<Db>)
      • + - Db
        • Represents a database, handling users and collections
        • + - close( )
          • Close the database and it's underlying connections
        • + - collection( name:string )
          • returns the collection with the given name
      • + - Collection

        • Represents a collection of data, handling insertion, lookup, modification and deletion

          insertOne ( doc: Object, callback: MongoCallback<InsertOneWriteOpResult>): void

          + - find ( query: Object ) : Cursor

          • finds all data matching the query object
          • returns a cursor for iteration over results

          + - findOne ( query: Object, callback: MongoCallback<Object> ) : void

          • finds the first data matching the query object
          • then calls the callback function and passes the object found

          + - updateOne ( query: Object, update: Object, options: ReplaceOneOptions,
          callback: MongoCallback<UpdateWriteOpResult>): void;

          • updates one document in the database
          • the first found with the given query
          • using the pattern given as the update-object
          • and the replace options

          + - remove( query: Object, callback: MongoCallback<WriteOpResult>): void;

          • removes all data matching the query object
      • + - Cursor
        • Represents a subset of data as a result of a find-command
        • + - close ( callback: MongoCallback<CursorResult> ): void
          • close the cursor
        • + - sort ( keyOrList: string | Object[] | Object, direction?: number): Cursor;
          • sort the cursor data
          • according to the criteria given as keyOrList
          • and a- or descendingas given in direction
        • + - toArray ( callback: MongoCallback<any[]> ): void;
          • creates a JavaScript-Array of Objects from the Cursor
      • + - query examples
        • + - { "key": "John" }
          • the value of property key must match the string "John"
        • + - { "key": { $gt: 20, $lt: 80, $ne: 50 } }
          • the value of property key must be
          • greater than 20, less than 80 and not equal to 50
        • + - { $or: [ { "key": { $lt: 20 } }, { "key": { $gt: 80 } }, { "key": 50 } ] }
          • the value of property key must be
          • less than 20 or greater than 80 or equal to 50
        • + - { $where: { "this.key1 + this.key2 > 100" } }
          • evaluates a JavaScript-expression
          • example: the sum of the properties key1 and key2 must exceed 100
          • this is mandatory to access the properties
      • + - update examples
        • + - { $set: { "key1": 10, "key2": "John", ... } }
          • sets the value of property key1 to 10, and key2 to "John"
        • + - { $unset: { "key": "", ... } }
          • removes the property key and its value
          • the given value is irrelevant
        • + - { $rename: { "key1": "key2" } }
          • renames property "key1" to "key2"
      • + - options example
        • + - { upsert: true }
          • if there is no data matching the query, it's inserted
